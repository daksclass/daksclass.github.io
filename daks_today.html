  • Computers are all about repetition. They make mundane stuff easy for us! Loops are a great way to make a computer do a repetative task.
  • A while loop is a lot like an if statement, it checks a condition, and if it's true, it runs! The big difference if that it doesn't stop running, it just goes and goes until the condition is no longer true.
import java.util.Random; import java.util.Scanner; public class Loopy { public static void main(String[] args) { Scanner input = new Scanner(System.in); Random gen = new Random(); int keepGoing = 1; while(keepGoing == 1){ System.out.println("Keep going?"); String answer = input.nextLine(); if(answer.equalsIgnoreCase("yes")){ System.out.println("Yay!"); } else if(answer.equalsIgnoreCase("no")){ System.out.println("Ok."); keepGoing = 0; } else{ System.out.println("Lets keep going!"); } } } }
